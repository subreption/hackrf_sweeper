436static inline void hackrf_sweep_state_set(hackrf_sweep_state_t* state, unsigned int flag) {
437 if (flag == SWEEP_STATE_STOPPED || flag == SWEEP_STATE_RUNNING) {
438 state->flags &= ~(SWEEP_STATE_STOPPED | SWEEP_STATE_RUNNING);
439 state->flags |= flag;
440 } else if (flag == SWEEP_STATE_EXITING) {
441 state->flags |= SWEEP_STATE_EXITING;
442 }
443}
444
445static inline void set_state_condition(hackrf_sweep_state_t* state, unsigned int flag) {
446 state->flags &= ~(SWEEP_STATE_INITIALIZED | SWEEP_STATE_RELEASED);
447 state->flags |= flag;
448}

450static inline void set_sweep_started(hackrf_sweep_state_t* state) {
451 state->flags |= SWEEP_STATE_SWEEP_STARTED;
452}
453
454static inline void clear_sweep_started(hackrf_sweep_state_t* state) {
455 state->flags &= ~SWEEP_STATE_SWEEP_STARTED;
456}

458static inline void set_sweep_finiteness(hackrf_sweep_state_t* state, unsigned int flag) {
